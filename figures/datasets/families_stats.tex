\documentclass[tikz]{standalone}

\usepackage{pgfplots}
\usepackage{pgfplotstable}

\begin{document}
    \pgfplotstableread[col sep=comma,header=false]{
        Unqualifiable, 62, 1947
        Building errors, 1420, 589
        Facet errors, 1648, 361
    }\foneEl
    \pgfplotstablecreatecol[
        create col/expr={
            \thisrow{1} + \thisrow{2}
        }
    ]{sum}{\foneEl}
    
    \pgfplotstableread[col sep=comma,header=false]{
        Unqualifiable, 56, 692
        Building errors, 435, 313
        Facet errors, 568, 180
    }\foneNa
    \pgfplotstablecreatecol[
        create col/expr={
            \thisrow{1} + \thisrow{2}
        }
    ]{sum}{\foneNa}

    \pgfplotstableread[col sep=comma,header=false]{
        Unqualifiable, 23, 455
        Building errors, 303, 175
        Facet errors, 411, 67
    }\fonePa
    \pgfplotstablecreatecol[
        create col/expr={
            \thisrow{1} + \thisrow{2}
        }
    ]{sum}{\fonePa}

    \pgfplotsset{
        percentage plot/.style={
            point meta=explicit,
            every node near coord/.append style={
                xshift=14pt,
                yshift=8pt,
                anchor=east,
                color=black,
                align=center,
                text width=.85cm
            },
            nodes near coords={
                \pgfmathtruncatemacro\iszero{\originalvalue==0}
                \ifnum\iszero=0
                    \pgfmathprintnumber[fixed zerofill,precision=0]{\pgfplotspointmeta}
                \fi
            },
            nodes near coords align={vertical},
            yticklabel=\pgfmathprintnumber{\tick}\,$\%$,
            visualization depends on={y \as \originalvalue}
        },
        percentage series/.style={
            table/y expr=(\thisrow{#1} / \thisrow{sum} * 100),
            table/meta=#1
        }
    }

    \begin{tikzpicture}
        \begin{axis}[
            ylabel={Ratio},
            ybar,
            height=8cm,
            width=14cm,
            enlarge x limits=0.2,
            percentage plot,
            bar width=.7cm,
            symbolic x coords={Unqualifiable, Building errors, Facet errors},
            xtick=data,
            xtick pos=left,
            ytick pos=left,
            x tick label style={font=\ttfamily, text width=2cm, align=center},
            ymax=100,
            ymin=0,
            legend style={
                legend columns=-2,
                at={(0.5,.99)},
                anchor=north
            }
        ]
            \addplot table [percentage series=1] {\foneEl};
            \addplot table [percentage series=1] {\foneNa};
            \addplot table [percentage series=1] {\fonePa};
            \legend{\strut Elancourt, \strut Nantes, \strut Paris-13}
        \end{axis}
    \end{tikzpicture}
\end{document}
